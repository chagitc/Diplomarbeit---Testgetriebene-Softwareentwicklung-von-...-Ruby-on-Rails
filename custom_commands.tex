\newcommand{\shadebox}[1]{%
\vspace{1\baselineskip}%
\noindent%
\begin{tikzpicture}%
\node at (0,0) [
   anchor=north west, 
    rounded corners,
   minimum width={\linewidth-1ex},
   draw=B4,
   fill=B4,
   copy shadow={
             shadow xshift=1ex,
             shadow yshift=-1ex,fill=S10,draw=white}
  ] 
{\parbox{0.9\linewidth}{#1}};
\end{tikzpicture}\vspace{1\baselineskip}}
